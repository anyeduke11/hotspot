# Hotspot v1.7 详细开发计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 Hotspot 从"信息平台"升级为"主动认知协作者"，补齐信息 → 知识 → 行动链路，落地 Hotspot ↔ Agent 双向环、SAG 生命周期、OKF + LLM-Wiki 2.0、KV 缓存层与六种认知模式。

**Architecture:** 在现有 FastAPI + SQLite + React 基线上新增 10 张表与 36 个 API 端点；通过 `backend/services/*_service.py` 承载业务规则，`backend/repository/*_repo.py` 封装数据访问，`backend/scheduler/jobs.py` 调度新 job，`agent/` 包实现 Agent CLI；前端按六种认知模式扩展路由与组件。

**Tech Stack:** Python 3.11, FastAPI, SQLite (FTS5), APScheduler, React 18, TypeScript, Vitest, pytest, watchdog, Pydantic.

```python
from __future__ import annotations
import json
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Optional
from backend.repository.db import get_connection

@dataclass
class Tag:
    id: str
    label: str
    type: str
    parent_id: Optional[str] = None
    weight: float = 1.0
    created_at: str = ""

class TagRepository:
    def add(self, id: str, label: str, type: str,
            parent_id: Optional[str] = None, weight: float = 1.0) -> Tag:
        now = datetime.now(timezone.utc).isoformat()
        conn = get_connection()
        conn.execute(
            "INSERT OR REPLACE INTO tags (id, label, type, parent_id, weight, created_at) VALUES (?, ?, ?, ?, ?, ?)",
            (id, label, type, parent_id, weight, now),
        )
        return self.get(id)

    def get(self, id: str) -> Optional[Tag]:
        row = get_connection().execute("SELECT * FROM tags WHERE id=?", (id,)).fetchone()
        if not row:
            return None
        return Tag(**dict(row))

    def list(self, type: Optional[str] = None, parent_id: Optional[str] = None,
             limit: int = 1000) -> list[Tag]:
        sql = "SELECT * FROM tags WHERE 1=1"
        params: list = []
        if type:
            sql += " AND type=?"
            params.append(type)
        if parent_id is not None:
            sql += " AND parent_id=?"
            params.append(parent_id)
        sql += " ORDER BY weight DESC, label LIMIT ?"
        params.append(limit)
        rows = get_connection().execute(sql, params).fetchall()
        return [Tag(**dict(r)) for r in rows]

    def suggest(self, q: str, limit: int = 10) -> list[Tag]:
        rows = get_connection().execute(
            "SELECT * FROM tags WHERE id LIKE ? OR label LIKE ? ORDER BY weight DESC LIMIT ?",
            (f"%{q}%", f"%{q}%", limit),
        ).fetchall()
        return [Tag(**dict(r)) for r in rows]
```

```python
from datetime import datetime, timezone
from typing import Optional
from backend.repository.db import get_connection

@dataclass
class ReadingState:
    entity_type: str
    entity_id: str
    opened_count: int
    dwell_ms: int
    last_read_at: Optional[str]
    created_at: str
    updated_at: str

class ReadingStateRepository:
    def get(self, entity_type: str, entity_id: str) -> Optional[ReadingState]:
        row = get_connection().execute(
            "SELECT * FROM reading_states WHERE entity_type=? AND entity_id=?",
            (entity_type, entity_id),
        ).fetchone()
        return ReadingState(**dict(row)) if row else None

    def record_open(self, entity_type: str, entity_id: str) -> None:
        now = datetime.now(timezone.utc).isoformat()
        get_connection().execute(
            """INSERT INTO reading_states (entity_type, entity_id, opened_count, dwell_ms, last_read_at, created_at, updated_at)
               VALUES (?, ?, 1, 0, ?, ?, ?)
               ON CONFLICT(entity_type, entity_id) DO UPDATE SET
               opened_count = opened_count + 1,
               last_read_at = excluded.last_read_at,
               updated_at = excluded.updated_at""",
            (entity_type, entity_id, now, now, now),
        )
```

```python
from __future__ import annotations
import json
import re
from pathlib import Path
from typing import Any

BASE_DIR = Path(__file__).resolve().parent.parent
RULES_PATH = BASE_DIR / "data" / "tag_rules.json"

CATEGORY_DOMAIN_MAP = {
    "ai": ["ai-security", "llm", "langchain"],
    "security": ["cve", "vulnerability", "network-security"],
}

def _load_rules() -> list[dict]:
    if not RULES_PATH.exists():
        return []
    return json.loads(RULES_PATH.read_text(encoding="utf-8")).get("rules", [])

def _regex_extract(text: str, rules: list[dict]) -> list[dict]:
    out = []
    for rule in rules:
        pattern = rule.get("pattern")
        if pattern and re.search(pattern, text, re.IGNORECASE):
            out.append({"tag_id": rule["tag_id"], "confidence": rule.get("confidence", 1.0)})
    return out

def _keyword_extract(text: str, rules: list[dict]) -> list[dict]:
    out = []
    lower = text.lower()
    for rule in rules:
        keywords = rule.get("keywords", [])
        if any(kw.lower() in lower for kw in keywords):
            out.append({"tag_id": rule["tag_id"], "confidence": rule.get("confidence", 0.7)})
    return out

def _category_domain_extract(text: str, title: str, category: str) -> list[dict]:
    out = []
    for tag_id in CATEGORY_DOMAIN_MAP.get(category, []):
        out.append({"tag_id": tag_id, "confidence": 0.5})
    return out

def extract_tags(text: str, title: str = "", category: str = "") -> list[dict]:
    rules = _load_rules()
    results = []
    results.extend(_regex_extract(text, rules))
    results.extend(_keyword_extract(text, rules))
    results.extend(_category_domain_extract(text, title, category))
    # 去重，取最高置信度
    merged: dict[str, float] = {}
    for r in results:
        tid = r["tag_id"]
        merged[tid] = max(merged.get(tid, 0.0), r["confidence"])
    return [{"tag_id": k, "confidence": v} for k, v in merged.items()]
```

```python
def list_by_tags(self, tag_ids: list[str], mode: str = "or", limit: int = 50, cursor: Optional[str] = None):
    # mode='and' 要求 hotspot 拥有全部 tag_ids
    # mode='or'  要求 hotspot 拥有任一 tag_ids
    placeholders = ",".join("?" * len(tag_ids))
    if mode == "and":
        sql = f"""
        SELECT h.* FROM hotspots h
        JOIN hotspot_tags ht ON h.id = ht.hotspot_id
        WHERE ht.tag_id IN ({placeholders})
        GROUP BY h.id
        HAVING COUNT(DISTINCT ht.tag_id) = ?
        ORDER BY h.ingested_at DESC
        LIMIT ?
        """
        params = tag_ids + [len(tag_ids), limit]
    else:
        sql = f"""
        SELECT DISTINCT h.* FROM hotspots h
        JOIN hotspot_tags ht ON h.id = ht.hotspot_id
        WHERE ht.tag_id IN ({placeholders})
        ORDER BY h.ingested_at DESC
        LIMIT ?
        """
        params = tag_ids + [limit]
    rows = get_connection().execute(sql, params).fetchall()
    return [HotspotItem.from_row(r) for r in rows]
```

```python
def sm2_schedule(grade: int, easiness: float, interval: int, repetitions: int):
    if grade < 3:
        repetitions = 0
        interval = 1
    else:
        if repetitions == 0:
            interval = 1
        elif repetitions == 1:
            interval = 6
        else:
            interval = round(interval * easiness)
        repetitions += 1
    easiness = max(1.3, easiness + 0.1 - (5 - grade) * (0.08 + (5 - grade) * 0.02))
    return easiness, interval, repetitions
```

```python
def evaluate_condition(condition: dict, hotspot: dict) -> bool:
    if condition.get("type") == "tag_match":
        tags = set(json.loads(hotspot.get("tags", "[]")))
        op = condition.get("operator", "OR")
        sub = condition.get("conditions", [])
        if op == "AND":
            return all(_match_sub(c, tags) for c in sub)
        return any(_match_sub(c, tags) for c in sub)
    return False

def _match_sub(c: dict, tags: set) -> bool:
    vals = set(c.get("value", []))
    if c.get("op") == "contains_any":
        return bool(tags & vals)
    return False
```

```python
def unified_search(q: str, sources: Optional[list[str]] = None, limit: int = 20) -> dict:
    source_filter = ""
    params = [q]
    if sources:
        source_filter = "AND source_type IN (" + ",".join("?" * len(sources)) + ")"
        params.extend(sources)
    params.append(limit)
    sql = f"""
    SELECT source_type, entity_id, title, source, category, ts, rank
    FROM unified_fts
    WHERE unified_fts MATCH ? {source_filter}
    ORDER BY rank DESC
    LIMIT ?
    """
    rows = get_connection().execute(sql, params).fetchall()
    grouped = {}
    for r in rows:
        grouped.setdefault(r["source_type"], []).append(dict(r))
    return {"query": q, "grouped": grouped}
```

```python
            (key, datetime.now(timezone.utc).isoformat()),

    def set(self, key: str, value: dict, etag: str = "", expires_seconds: int = 60):

            (key, json.dumps(value), etag, expires, now, now),
        )

    def invalidate_prefix(self, prefix: str):
```

```python
import time
from agent.client import HotspotClient
from agent.executor import execute_task

class AgentPoller:
    def __init__(self, client: HotspotClient):
        self.client = client
        self.interval = 300

    def run(self):
        config = self.client.get_config()
        self.interval = config.get("collect_interval_seconds", 300)
        while True:
            tasks = self.client.get_tasks()
            if tasks:
                self.interval = max(60, self.interval // 2)
                for task in tasks:
                    result = execute_task(task)
                    self.client.complete_task(task["task_id"], "done", result)
            else:
                self.interval = min(600, self.interval + 30)
            time.sleep(self.interval)
```
